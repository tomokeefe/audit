**Change Share Behavior:**
```tsx
// Current: Uses Web Share API with clipboard fallback
onClick={() => {
  if (navigator.share) {
    navigator.share({...});
  } else {
    navigator.clipboard.writeText(window.location.href);
    alert("Link copied to clipboard!");
  }
}}

// Alternative: Email share
onClick={() => {
  window.location.href = `mailto:?subject=Brand Audit Report&body=${window.location.href}`;
}}
```
